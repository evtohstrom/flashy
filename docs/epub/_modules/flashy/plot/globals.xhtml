[docs]def customFormatter(factor, prec=1, width=2):
    """create a mpl formatter whith specialized format 
    for plotting labels:
    width(prec) x 10^factor.
    """
    fstr = '{:{width}.{prec}f}'
    exp = 10.0**factor
    return FuncFormatter(lambda x, pos:fstr.format(x/exp, 
                         width=width, prec=prec))
